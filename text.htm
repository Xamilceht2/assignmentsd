# Open the input file and output file
with open('paste-2.txt', 'r') as infile, open('times_and_years.txt', 'w') as outfile:
    for line in infile:
        # Split each line into columns based on whitespace
        columns = line.split()
        if len(columns) > 8:  # Ensure the line has enough columns to extract time and year
            time = columns[1]   # The time is in the second column (index 1)
